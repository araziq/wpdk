299 function wpdk_delta_object( $last_version, $old_version ) {
300 
301  $last_version_stack = array();
302  $old_version_stack = array();
303 
304  /* Creo un elenco di tutte le proprietà della classe $old_version. Questo elenco mi indica il nome della
305  proprietà e il tipo.
306  */
307  foreach ( $old_version as $key => $value ) {
308  $old_version_stack[$key] = $value;
309  }
310 
311  /* Ora ciclo nella versione recente */
312  foreach ( $last_version as $key => $value ) {
313  /* Se la precedente versione non contiene la proprietà di quella nuova, la imposto con il valore di default. */
314  if ( !isset( $old_version_stack[$key] ) ) {
315  $old_version->$key = $value;
316  }
317 
318  elseif ( empty( $old_version_stack[$key] ) || is_null( $old_version_stack[$key] ) ) {
319  $old_version->$key = $value;
320  }
321 
322  /* La proprietà esiste. */
323  else {
324  /* Se la proprietà c'è potrebbe essere a sua volta un oggeto, quindi controllo ed eventualmente ciclo su
325  questo.
326  */
327  if ( is_object( $value ) ) {
328  wpdk_delta_object( $value, $old_version->$key );
329  }
330  }
331  }
332 
333  /* Precedentemente abbiamo controllato per 'mancanze' nella vecchia classe, ora facciamo un controllo speculare
334  cioè verifichiamo che la nuova struttura non abbia eliminato qualcosa.
335  Come nel caso precedente creo un elenco delle proprietà dell'ultima versione.
336  */
337  foreach ( $last_version as $key => $value ) {
338  $last_version_stack[$key] = $value;
339  }
340 
341  /* Ora ciclo nella vecchia versione */
342  foreach ( $old_version as $key => $value ) {
343  /* Se non esiste più questa proprietà... */
344  if ( !isset( $last_version_stack[$key] ) ) {
345  /* La elimino */
346  unset( $old_version->$key );
347  }
348  }
349  /* Ok, $old_version ora è allineata. */
350  return $old_version;
351 }
