303 function wpdk_delta_object( $last_version, $old_version ) {
304 
305  $last_version_stack = array();
306  $old_version_stack = array();
307 
308  /* Creo un elenco di tutte le proprietà della classe $old_version. Questo elenco mi indica il nome della
309  proprietà e il tipo.
310  */
311  foreach ( $old_version as $key => $value ) {
312  $old_version_stack[$key] = $value;
313  }
314 
315  /* Ora ciclo nella versione recente */
316  foreach ( $last_version as $key => $value ) {
317  /* Se la precedente versione non contiene la proprietà di quella nuova, la imposto con il valore di default. */
318  if ( !isset( $old_version_stack[$key] ) ) {
319  $old_version->$key = $value;
320  }
321 
322  elseif ( empty( $old_version_stack[$key] ) || is_null( $old_version_stack[$key] ) ) {
323  $old_version->$key = $value;
324  }
325 
326  /* La proprietà esiste. */
327  else {
328  /* Se la proprietà c'è potrebbe essere a sua volta un oggeto, quindi controllo ed eventualmente ciclo su
329  questo.
330  */
331  if ( is_object( $value ) ) {
332  wpdk_delta_object( $value, $old_version->$key );
333  }
334  }
335  }
336 
337  /* Precedentemente abbiamo controllato per 'mancanze' nella vecchia classe, ora facciamo un controllo speculare
338  cioè verifichiamo che la nuova struttura non abbia eliminato qualcosa.
339  Come nel caso precedente creo un elenco delle proprietà dell'ultima versione.
340  */
341  foreach ( $last_version as $key => $value ) {
342  $last_version_stack[$key] = $value;
343  }
344 
345  /* Ora ciclo nella vecchia versione */
346  foreach ( $old_version as $key => $value ) {
347  /* Se non esiste più questa proprietà... */
348  if ( !isset( $last_version_stack[$key] ) ) {
349  /* La elimino */
350  unset( $old_version->$key );
351  }
352  }
353  /* Ok, $old_version ora è allineata. */
354  return $old_version;
355 }
