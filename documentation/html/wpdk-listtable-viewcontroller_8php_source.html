744  {

754  );
755  $href = add_query_arg( $args );
756  $actions[$key] = sprintf( '<a href="%s">%s</a>', $href, $label );
757  }
758 
759  if ( empty( $status ) || $status != 'trash' ) {
760  unset( $actions['untrash'] );
761  unset( $actions['delete'] );
762  }
763  else if ( $status == 'trash' ) {
764  unset( $actions['edit'] );

766  }
